//Registrar Usuario
export const registrarUsuario = async (req, res) => {
    const { name, email, password } = req.body;
    const TIPO_USUARIO_DEFAULT = 2;
    try {
        const existe = await pool.query(
            'SELECT * FROM users WHERE email = $1',
            [email]
        );
        if (existe.rows.length > 0) {
